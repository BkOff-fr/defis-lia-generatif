# C33.4 Doc interactive (Astro Content Collections + Pagefind) — Plan

> **For agentic workers:** REQUIRED SUB-SKILL: superpowers:executing-plans.

**Goal:** Exposer la documentation du repo (`docs/*.md` + `docs/adr/*.md`) sous `/docs/*` et `/adrs/*` du site, avec sidebar de navigation, fil d'Ariane, "Edit on GitHub", recherche full-text Pagefind, sans dupliquer le contenu — single source of truth.

**Architecture :**
- **Sync script** : `scripts/sync-docs.mjs` (Node, cross-platform Windows/Linux). Copie `docs/*.md` (sauf `docs/adr/`) → `site/src/content/docs/` ; `docs/adr/*.md` → `site/src/content/adrs/`. Préserve la structure des sous-dossiers (`methodology/`, `personas/`, etc.). Conserve le chemin source en injectant un front-matter minimal `sourcePath` au sommet de chaque fichier copié (pour générer "Edit on GitHub").
- **Astro 5 Content Layer** : `src/content.config.ts` avec `defineCollection({ loader: glob(...), schema: z.object({ ... }) })`. Tous les champs optionnels — fallback first H1 + auto-excerpt.
- **DocLayout** : sidebar gauche (tree groupé par dossier) + contenu central + breadcrumb + "Edit on GitHub" footer.
- **Pagefind** : installé en dev, indexation post-build sur `dist/docs/**` + `dist/adrs/**`. SearchBar Svelte 5 dans Topbar.
- **Rendu** : Astro markdown natif → Shiki built-in pour syntax highlight (pas besoin de composant CodeBlock custom).
- **Hors scope** : Callout custom (aucun .md ne l'utilise) → différé tant qu'aucun MD ne le demande.

**Tech additions :**
- `pagefind` (déjà installé en C33.1 dans les deps).
- Pas d'autre dépendance.

**Wordings — défauts UI :**
- Sidebar header : « Documentation »

- [ ] **Step 1.1 — Write `scripts/sync-docs.mjs`**

```js
#!/usr/bin/env node
/**
 * Synchronise docs/ → site/src/content/.
 * - docs/*.md (sauf docs/adr/) → site/src/content/docs/<path>
 * - docs/adr/*.md → site/src/content/adrs/<filename>
 * Injecte un front-matter `sourcePath` (chemin relatif au repo root) si
 * le fichier n'a pas déjà de front-matter, pour générer "Edit on GitHub".
 *
 * Exécutable depuis n'importe où ; on calcule les chemins relatifs au
 * dossier qui contient ce script (scripts/).
 */
import { mkdirSync, readdirSync, readFileSync, writeFileSync, statSync, rmSync, existsSync } from 'node:fs';
import { dirname, join, relative, resolve } from 'node:path';
import { fileURLToPath } from 'node:url';

const __filename = fileURLToPath(import.meta.url);
const __dirname = dirname(__filename);
const REPO_ROOT = resolve(__dirname, '..');
const DOCS_DIR = resolve(REPO_ROOT, 'docs');
const TARGET_DOCS = resolve(REPO_ROOT, 'site/src/content/docs');
const TARGET_ADRS = resolve(REPO_ROOT, 'site/src/content/adrs');

const HAS_FRONTMATTER = /^---\r?\n/;

function walk(dir) {
  const entries = readdirSync(dir, { withFileTypes: true });
  const out = [];
  for (const e of entries) {
    const full = join(dir, e.name);
    if (e.isDirectory()) {
      out.push(...walk(full));
    } else if (e.isFile() && e.name.endsWith('.md')) {
      out.push(full);
    }
  }
  return out;
}

function syncFile(src, dest) {
  mkdirSync(dirname(dest), { recursive: true });
  const raw = readFileSync(src, 'utf8');
  const sourcePath = relative(REPO_ROOT, src).replace(/\\/g, '/');
  let out;
  if (HAS_FRONTMATTER.test(raw)) {
    // Le fichier a déjà du front-matter — on injecte sourcePath dans le bloc.
    out = raw.replace(/^---\r?\n/, `---\nsourcePath: ${sourcePath}\n`);
  } else {
    // Pas de front-matter — on en crée un minimal.
    out = `---\nsourcePath: ${sourcePath}\n---\n\n${raw}`;
  }
  writeFileSync(dest, out, 'utf8');
}

function clean(dir) {
  if (existsSync(dir)) {
    rmSync(dir, { recursive: true, force: true });
  }
  mkdirSync(dir, { recursive: true });
}

function main() {
  if (!existsSync(DOCS_DIR)) {
    console.error(`Source docs/ not found at ${DOCS_DIR}`);
    process.exit(1);
  }

  clean(TARGET_DOCS);
  clean(TARGET_ADRS);

  let count = 0;
  const all = walk(DOCS_DIR);
  for (const file of all) {
    const relPath = relative(DOCS_DIR, file).replace(/\\/g, '/');
    if (relPath.startsWith('adr/')) {
      const flat = relPath.replace(/^adr\//, '');
      syncFile(file, join(TARGET_ADRS, flat));
    } else if (relPath.startsWith('superpowers/')) {
      // Skip internal planning docs — not part of the public guide.
      continue;
    } else {
      syncFile(file, join(TARGET_DOCS, relPath));
    }
    count++;
  }

  console.log(`✓ Synced ${count} markdown files to site/src/content/`);
}

main();
```

- [ ] **Step 2.1 — Write `site/src/lib/doc-tree.ts`**

```ts
interface DocEntry {
  id: string;
  data: { title?: string; sourcePath: string; order?: number; category?: string };
  body?: string;
}

export interface TreeNode {
  type: 'folder' | 'leaf';
  name: string;
  href?: string;
  title: string;
  children?: TreeNode[];
}

/**
 * Extract the first H1 from a markdown body to use as title fallback.
 */
export function firstH1(body: string | undefined): string | null {
  if (!body) return null;
  const m = body.match(/^#\s+(.+)$/m);
  return m ? m[1].trim() : null;
}

export function getEntryTitle(entry: DocEntry): string {
  if (entry.data.title) return entry.data.title;
  const h1 = firstH1(entry.body);
  if (h1) return h1;
  // Fallback: humanize the id ("methodology/glossaire" → "Glossaire")
  const last = entry.id.split('/').pop() || entry.id;
  return last
    .replace(/[-_]/g, ' ')
    .replace(/\.md$/i, '')
    .replace(/\b\w/g, (c) => c.toUpperCase());
}

/**
 * Build a sidebar tree from a flat list of doc entries.
 * Folders come from the id path segments (e.g. "methodology/foo" → folder "methodology" leaf "foo").
 */
export function buildTree(entries: DocEntry[], baseHref: string): TreeNode[] {
  const folders: Record<string, TreeNode> = {};
  const root: TreeNode[] = [];

  const sorted = [...entries].sort((a, b) => {
    const ao = a.data.order ?? 999;
    const bo = b.data.order ?? 999;
    if (ao !== bo) return ao - bo;
    return getEntryTitle(a).localeCompare(getEntryTitle(b));
  });

  for (const entry of sorted) {
    const segments = entry.id.split('/');
    if (segments.length === 1) {
      root.push({
        type: 'leaf',
        name: segments[0],
        href: `${baseHref}/${entry.id}/`,
        title: getEntryTitle(entry),
      });
    } else {
      const folderName = segments[0];
      if (!folders[folderName]) {
        folders[folderName] = {
          type: 'folder',
          name: folderName,
          title: folderName.replace(/[-_]/g, ' ').replace(/\b\w/g, (c) => c.toUpperCase()),
          children: [],
        };
        root.push(folders[folderName]);
      }
      folders[folderName].children!.push({
        type: 'leaf',
        name: segments.slice(1).join('/'),
        href: `${baseHref}/${entry.id}/`,
        title: getEntryTitle(entry),
      });
    }
  }
  return root;
}
```

<script lang="ts">
  import { onMount } from 'svelte';

  let open = $state(false);
  let containerEl: HTMLDivElement | null = $state(null);
  // eslint-disable-next-line @typescript-eslint/no-explicit-any
  let pagefindUI: any = null;

  async function ensurePagefind() {
    if (pagefindUI || !containerEl) return;
    // Pagefind UI script is served from /pagefind/pagefind-ui.js after build.
    // We attach it directly into our container — Pagefind handles the rest.
    await import(/* @vite-ignore */ '/pagefind/pagefind-ui.js').catch(() => {
      console.warn('Pagefind UI not yet built — run `npm run build` first.');
    });
    // @ts-expect-error global PagefindUI from script
    if (typeof window.PagefindUI === 'function') {
      // @ts-expect-error global
      pagefindUI = new window.PagefindUI({
        element: containerEl,
        showSubResults: true,
        translations: {
          placeholder: 'Rechercher dans la documentation…',
          clear_search: 'Effacer',
          load_more: 'Voir plus',
          search_label: 'Rechercher',
          filters_label: 'Filtres',

          search_suggestion: 'Aucun résultat pour [SEARCH_TERM]. Suggestions :',
          searching: 'Recherche…',
        },
      });
    }
  }

  function toggle() {
    open = !open;
